// register Metro design fonts
import "com.sweaxizone.metro/fonts";

// third-party
import { createRoot } from "react-dom/client";
import * as React from "react";
import {
  Root,
  Group,
  HGroup,
  VGroup,
  Button,
  Label,
  Icon,
  BlackoutModal,
} from "com.sweaxizone.metro/components";
import { RTLProvider } from "com.sweaxizone.metro/layout";
import {
  Primary,
  ThemePresets,
  ThemeProvider,
  type Theme,
} from "com.sweaxizone.metro/theme";

/**
 * The test.
 */
function App() {
  const [modal_open, set_modal_open] = React.useState<boolean>(false);

  return (
    <ThemeProvider theme={ThemePresets.get("dark")}>
      <RTLProvider rtl={false}>
        <Primary prefer={false}>
          <Root
            full
            solid
            selection={false}
            style={{
              overflowY: "auto",
            }}>
            <VGroup padding={10} gap={10}>
              <Button outline click={e=>{set_modal_open(true)}}>Open modal</Button>
              {
                modal_open ?
                  <BlackoutModal close={()=>{set_modal_open(false)}}>
                    <VGroup solid style={{flexGrow: "1"}} padding={10} paddingTop={64} paddingBottom={64}>
                      <VGroup horizontalAlign="right">
                        <HGroup gap={10}>
                          <Button outline click={()=>{set_modal_open(false)}}>Close</Button>
                        </HGroup>
                      </VGroup>
                    </VGroup>
                  </BlackoutModal> :
                  undefined
              }
            </VGroup>
          </Root>
        </Primary>
      </RTLProvider>
    </ThemeProvider>
  );
}

// Render App
const root = createRoot(document.getElementById("root")!);
root.render(<App />);
